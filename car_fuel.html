# Define bins for City Fuel Economy
bins = [0, 10, 20, 30, 40, 50, 60]
labels = ['0-10', '11-20', '21-30', '31-40', '41-50', '51-60']

# Categorize City Fuel Economy into bins
data['City FE Category'] = pd.cut(data['City FE'], bins=bins, labels=labels)

# Calculate counts for each category
city_fe_counts = data['City FE Category'].value_counts()

# Plotting the pie chart
plt.figure(figsize=(8, 8))
plt.pie(city_fe_counts, labels=city_fe_counts.index, autopct='%1.1f%%', startangle=140)
plt.title('Distribution of City Fuel Economy Categories')
plt.axis('equal')  # Equal aspect ratio ensures that pie is drawn as a circle.
plt.legend()
plt.show()
